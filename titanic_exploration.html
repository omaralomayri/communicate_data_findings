#function calculates Survival rate and creates a barplot for Survival rate

def rate_of_survive(column):
    
    display_pivot_table(column) #show pivot table for input column
    print('\n') #print new line
    for cate in titanic_df[column].value_counts().index: # for loop for uniques value inside column
        category=titanic_df[titanic_df[column]==cate]["Survived"] #sign variable for series of specific unique value 
        rate_first_class=round(sum(category)/len(category)*100) #calculate rate of survive 
        print(f'for {cate} the Survival rate is {rate_first_class}%') 
    print('\n')
    base_color=sns.color_palette()[0]
    sns.barplot(data=titanic_df,x=column,y='Survived',color=base_color) #create barplot
    plt.title('Survived Vs {}'.format(column))
    plt.show()
